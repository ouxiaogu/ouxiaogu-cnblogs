    $f(r,c)=f(-N,N)-(N-c), \qquad r=-N,r<c$  \\ % 北
    $f(r,c)=f(-(N-1),(N-1))+(r+(N-1))+1, \qquad c=N,r>-c$    \\ % 东
    $f(r,c)=f(N,-(N-1))-(c+(N-1)), \qquad r=N,r>c$    \\ % 南
    $f(r,c)=f(N,-(N-1))+(N-r)+1, \qquad c=-N,r<-c$    \\ % 西
